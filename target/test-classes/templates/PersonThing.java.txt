package dev.coffeezombie.wasp.entity;

import dev.coffeezombie.wasp.dto.PersonThingDto;
import jakarta.persistence.GeneratedValue;
import jakarta.persistence.GenerationType;
import jakarta.persistence.Id;
import lombok.Data;
import lombok.NoArgsConstructor;
import org.modelmapper.ModelMapper;

import java.Math.BigDecimal;

@Entity
@Data
@NoArgsConstructor
public class PersonThing {

    @Id
    @GeneratedValue(strategy = GenerationType.IDENTITY)
    private Long id;

    private Long myLong;
    private BigDecimal myBd;

    public PersonThing(PersonThingDto dto) {
        var mapper = new ModelMapper();
        return mapper.map(entity, PersonThing.class)
    }

}
